);
CREATE TABLE medicos (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,

    especialidade_id INT NOT NULL,
    FOREIGN KEY (especialidade_id) REFERENCES especialidades(id)
);
CREATE TABLE pacientes (
